The Math Behind It

# Vector addition in angle space
def add_angles(angle1, angle2):
    """Add two angles - equivalent to rotating one by the other"""
    return (angle1 + angle2) % (2 * np.pi)

def subtract_angles(angle1, angle2):
    """Find angular difference between two angles"""
    diff = angle1 - angle2
    # Handle wraparound
    if diff > np.pi:
        diff -= 2 * np.pi
    elif diff < -np.pi:
        diff += 2 * np.pi
    return diff
